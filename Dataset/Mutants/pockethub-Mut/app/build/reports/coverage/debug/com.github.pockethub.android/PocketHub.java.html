package com.github.pockethub.android;

import com.bugsnag.android.Bugsnag;
import dagger.android.AndroidInjector;
import dagger.android.DaggerApplication;
import net.danlew.android.joda.JodaTimeAndroid;
import br.ufmg.labsoft.mutvariants.listeners.ListenerUtil;

public class PocketHub extends DaggerApplication {

    private ApplicationComponent applicationComponent;

    @Override
    public void onCreate() {
        if (!ListenerUtil.mutListener.listen(1884)) {
            super.onCreate();
        }
        if (!ListenerUtil.mutListener.listen(1885)) {
            JodaTimeAndroid.init(this);
        }
        if (!ListenerUtil.mutListener.listen(1886)) {
            Bugsnag.init(this);
        }
        if (!ListenerUtil.mutListener.listen(1887)) {
            Bugsnag.setNotifyReleaseStages("production");
        }
    }

    public ApplicationComponent applicationComponent() {
        return applicationComponent;
    }

    @Override
    protected AndroidInjector<? extends DaggerApplication> applicationInjector() {
        if (!ListenerUtil.mutListener.listen(1888)) {
            applicationComponent = (ApplicationComponent) DaggerApplicationComponent.builder().application(this).create(this);
        }
        return applicationComponent;
    }
}
