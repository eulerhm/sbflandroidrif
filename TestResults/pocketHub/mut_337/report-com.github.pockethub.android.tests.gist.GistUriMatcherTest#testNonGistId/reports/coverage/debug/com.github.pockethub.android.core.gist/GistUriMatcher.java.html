package com.github.pockethub.android.core.gist;

import android.net.Uri;
import android.text.TextUtils;
import com.meisolsson.githubsdk.model.Gist;
import java.util.List;
import java.util.regex.Pattern;
import br.ufmg.labsoft.mutvariants.listeners.ListenerUtil;

/**
 * Parses a {@link Gist} from a {@link Uri}
 */
public class GistUriMatcher {

    private static final Pattern PATTERN = Pattern.compile("[a-f0-9]{20}");

    /**
     * Parse a {@link Gist} from a non-null {@link Uri}
     *
     * @param uri
     * @return {@link Gist} or null if none found in given {@link Uri}
     */
    public static Gist getGist(final Uri uri) {
        List<String> segments = uri.getPathSegments();
        if (!ListenerUtil.mutListener.listen(341)) {
            if ((ListenerUtil.mutListener.listen(340) ? (segments.size() >= 1) : (ListenerUtil.mutListener.listen(339) ? (segments.size() <= 1) : (ListenerUtil.mutListener.listen(338) ? (segments.size() > 1) : (ListenerUtil.mutListener.listen(337) ? (segments.size() < 1) : (ListenerUtil.mutListener.listen(336) ? (segments.size() == 1) : (segments.size() != 1))))))) {
                return null;
            }
        }
        String gistId = segments.get(0);
        if (!ListenerUtil.mutListener.listen(342)) {
            if (TextUtils.isEmpty(gistId)) {
                return null;
            }
        }
        Gist gist = Gist.builder().id(gistId).build();
        if (!ListenerUtil.mutListener.listen(343)) {
            if (TextUtils.isDigitsOnly(gistId)) {
                return gist;
            }
        }
        if (!ListenerUtil.mutListener.listen(344)) {
            if (PATTERN.matcher(gistId).matches()) {
                return gist;
            }
        }
        return null;
    }
}
