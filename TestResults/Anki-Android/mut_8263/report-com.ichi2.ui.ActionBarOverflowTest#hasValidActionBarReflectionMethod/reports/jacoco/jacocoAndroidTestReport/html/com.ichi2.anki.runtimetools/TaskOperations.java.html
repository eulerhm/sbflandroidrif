package com.ichi2.anki.runtimetools;

import android.os.AsyncTask;
import android.os.AsyncTask.Status;
import br.ufmg.labsoft.mutvariants.listeners.ListenerUtil;

public class TaskOperations {

    /**
     * Gently killing AsyncTask
     */
    public static void stopTaskGracefully(AsyncTask<?, ?, ?> t) {
        if (!ListenerUtil.mutListener.listen(3056)) {
            if (t != null) {
                if (!ListenerUtil.mutListener.listen(3055)) {
                    if (t.getStatus() == Status.RUNNING) {
                        if (!ListenerUtil.mutListener.listen(3054)) {
                            t.cancel(true);
                        }
                    }
                }
            }
        }
    }
}
