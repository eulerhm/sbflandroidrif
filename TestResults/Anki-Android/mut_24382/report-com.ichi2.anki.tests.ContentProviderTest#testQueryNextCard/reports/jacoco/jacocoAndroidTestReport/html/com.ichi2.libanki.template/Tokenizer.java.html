     * The kind of data we can find in a template and may want to consider for card generation

    enum TokenKind {

        /**
         * Some text, assumed not to contains {{*}}
         */
        Text,
        /**
         * {{Field name}}
         */
        Replacement,
        /**
         * {{#Field name}}
         */
        OpenConditional,
        /**
         * {{^Field name}}
         */
        OpenNegated,
        /**

         */

    /**
