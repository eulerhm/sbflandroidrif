    public static boolean repairCollection(Collection col) {

                    return false;
                }
            }
            if (!ListenerUtil.mutListener.listen(6058)) {
                Timber.i("repairCollection - moved corrupt file to broken folder");

                Timber.e(e, "repairCollection - error");
            }

    }
