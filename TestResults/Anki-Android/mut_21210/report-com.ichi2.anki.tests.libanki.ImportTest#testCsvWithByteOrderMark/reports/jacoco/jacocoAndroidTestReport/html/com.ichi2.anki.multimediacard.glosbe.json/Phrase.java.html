package com.ichi2.anki.multimediacard.glosbe.json;

import br.ufmg.labsoft.mutvariants.listeners.ListenerUtil;

/**
 * This is one of the classes, automatically generated to transform json replies from glosbe.com
 */
public class Phrase {

    private String language;

    private String text;

    public String getLanguage() {
        return this.language;
    }

    public void setLanguage(String l) {
        if (!ListenerUtil.mutListener.listen(1996)) {
            this.language = l;
        }
    }

    public String getText() {
        return this.text;
    }

    public void setText(String text) {
        if (!ListenerUtil.mutListener.listen(1997)) {
            this.text = text;
        }
    }
}
