package org.onebusaway.android.ui;

import org.onebusaway.android.R;
import org.onebusaway.android.util.UIUtils;
import android.content.Intent;
import android.net.Uri;
import android.os.Bundle;
import androidx.appcompat.app.AppCompatActivity;
import androidx.core.content.pm.ShortcutInfoCompat;
import androidx.core.content.pm.ShortcutManagerCompat;
import br.ufmg.labsoft.mutvariants.listeners.ListenerUtil;

public class MyRecentRoutesActivity extends AppCompatActivity {

    // 
    @Override
    public void onCreate(Bundle savedInstanceState) {
        if (!ListenerUtil.mutListener.listen(7)) {
            super.onCreate(savedInstanceState);
        }
        Intent myIntent = getIntent();
        if (!ListenerUtil.mutListener.listen(10)) {
            if (Intent.ACTION_CREATE_SHORTCUT.equals(myIntent.getAction())) {
                ShortcutInfoCompat shortcut = getShortcut();
                if (!ListenerUtil.mutListener.listen(8)) {
                    ShortcutManagerCompat.requestPinShortcut(this, shortcut, null);
                }
                if (!ListenerUtil.mutListener.listen(9)) {
                    setResult(RESULT_OK, shortcut.getIntent());
                }
            }
        }
        if (!ListenerUtil.mutListener.listen(11)) {
            finish();
        }
    }

    private ShortcutInfoCompat getShortcut() {
        final Uri uri = MyTabActivityBase.getDefaultTabUri(MyRecentRoutesFragment.TAB_NAME);
        return UIUtils.makeShortcutInfo(this, getString(R.string.recent_routes_shortcut), new Intent(this, MyRoutesActivity.class).setData(uri), R.drawable.ic_history);
    }
}
