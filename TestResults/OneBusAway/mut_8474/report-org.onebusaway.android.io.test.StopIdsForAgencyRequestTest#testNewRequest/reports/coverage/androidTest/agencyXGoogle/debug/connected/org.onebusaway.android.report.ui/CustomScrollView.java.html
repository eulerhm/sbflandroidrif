package org.onebusaway.android.report.ui;

import android.content.Context;
import android.graphics.Rect;
import android.util.AttributeSet;
import android.view.MotionEvent;
import android.view.View;
import android.widget.ScrollView;
import java.util.ArrayList;
import java.util.List;
import br.ufmg.labsoft.mutvariants.listeners.ListenerUtil;

/**
 * Created by Cagri Cetin
 */
public class CustomScrollView extends ScrollView {

    List<View> mInterceptScrollViews = new ArrayList<>();

    public CustomScrollView(Context context) {
        super(context);
    }

    public CustomScrollView(Context context, AttributeSet attrs) {
        super(context, attrs);
    }

    public CustomScrollView(Context context, AttributeSet attrs, int defStyle) {
        super(context, attrs, defStyle);
    }

    public void addInterceptScrollView(View view) {
        if (!ListenerUtil.mutListener.listen(10976)) {
            mInterceptScrollViews.add(view);
        }
    }

    @Override
    public boolean onInterceptTouchEvent(MotionEvent event) {
        if (!ListenerUtil.mutListener.listen(10985)) {
            // check if we have any views that should use their own scrolling
            if ((ListenerUtil.mutListener.listen(10981) ? (mInterceptScrollViews.size() >= 0) : (ListenerUtil.mutListener.listen(10980) ? (mInterceptScrollViews.size() <= 0) : (ListenerUtil.mutListener.listen(10979) ? (mInterceptScrollViews.size() < 0) : (ListenerUtil.mutListener.listen(10978) ? (mInterceptScrollViews.size() != 0) : (ListenerUtil.mutListener.listen(10977) ? (mInterceptScrollViews.size() == 0) : (mInterceptScrollViews.size() > 0))))))) {
                int x = (int) event.getX();
                int y = (int) event.getY();
                Rect bounds = new Rect();
                if (!ListenerUtil.mutListener.listen(10984)) {
                    {
                        long _loopCounter148 = 0;
                        for (View view : mInterceptScrollViews) {
                            ListenerUtil.loopListener.listen("_loopCounter148", ++_loopCounter148);
                            if (!ListenerUtil.mutListener.listen(10982)) {
                                view.getHitRect(bounds);
                            }
                            if (!ListenerUtil.mutListener.listen(10983)) {
                                if (bounds.contains(x, y)) {
                                    // were touching a view that should intercept scrolling
                                    return false;
                                }
                            }
                        }
                    }
                }
            }
        }
        return super.onInterceptTouchEvent(event);
    }
}
